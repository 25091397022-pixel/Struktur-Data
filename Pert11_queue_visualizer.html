def hot_potato(names, num): q = Queue() for name in names: q.enqueue(name) while len(q) > 1: for _ in range(num): q.enqueue(q.dequeue()) # oper melingkar q.dequeue() # ← tersingkir! return q.dequeue() # pemenang!
